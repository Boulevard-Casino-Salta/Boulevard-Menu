fetch('productos.json') .then(res => res.json()) .then(data => { const contenedor = document.getElementById('menu-lista'); // Esto crea una lista de categorías únicas como "BEBIDAS", "COMIDAS", etc. const categorias = [...new Set(data.map(p => p.cat))]; let htmlFinal = ''; categorias.forEach(cat => { // Agregamos un título para cada categoría htmlFinal += `

${cat}

`; // Filtramos los productos que pertenecen a esa categoría const filtrados = data.filter(p => p.cat === cat); filtrados.forEach(prod => { htmlFinal += `
${prod.nombre} $${prod.precio.toLocaleString('es-AR')}
`; }); }); contenedor.innerHTML = htmlFinal; });
